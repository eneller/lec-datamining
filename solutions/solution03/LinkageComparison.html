# The median instead of the centroid linkage is ussed because the latter is buggy (produces the same result for k=1 and k=2)
methods = ['average', 'weighted', 'median', 'complete', 'single']
methods_names = ['Group avg.', 'Unweighted avg.', 'Median', 'Complete', 'Single']

set_font()
fig, axes = plt.subplots(5, 5, figsize=(12, 12))

for i, method in enumerate(methods):
    # Apply the agglomerative algorithm to obtain the hierarchy information
    tree = linkage(data, method=method, optimal_ordering=True)
    
    for j, k in enumerate(range(1, 6)):
        # Undo the last merges to retrieve at most k clusters
        labels = fcluster(tree, k, criterion='maxclust')
        ax = axes[j, i]
        
        # Plot data points for each class
        for label in np.unique(labels):
            data_cluster = data[labels == label, :]
            ax.scatter(data_cluster[:, 0], data_cluster[:, 1])

        # No tick labels
        ax.tick_params(axis='x', bottom=False, top=False, labelbottom=False, which='both')
        ax.tick_params(axis='y', left=False, right=False, labelleft=False, which='both')
        
        # Top label
        if k == 1:
            ax.set_title(methods_names[i])

        # Label on the left side
        if method == 'average':
            ax.set_ylabel('$k = %d$' % k)

Part 3: Interpretation¶

  • single: the two clusters with the lowest minimal distance to each other are merged. In this case, the blue cluster is nearer to the orange cluster than to the green single point.
  • complete: the two clusters with the lowest maximal distance to each other are merged. In this case, the maximal distance between the points in orange and green is lowest compared to the other maximal distances.
